package packageDemo;

public class SSort implements iSorting {
	public void sort(int[] a) {
		int len = a.length;
		for (int i = 0; i < len - 1; i++) {
			// minimum element in array
			int min = i;
			for (int j = i + 1; j < len; j++)
				if (a[j] < a[min])
					min = j;

			// Swap minimum element with the first element
			int temp = a[min];
			a[min] = a[i];
			a[i] = temp;

		}
	}
}

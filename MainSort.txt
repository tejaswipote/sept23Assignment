package packageDemo;

import java.util.*;

public class MainSort {

	private static Scanner sc;

	static void printSortedData(int a[]) {
		for (int i = 0; i < a.length; i++) {
			System.out.println(a[i]);
		}
	}

	public static void main(String[] args) {
		sc = new Scanner(System.in);
		// int a[]= {1,2,5,4,3,8}; //initialized array in program itself

		System.out.println("Enter the size of array");
		int size = sc.nextInt();

		int[] a = new int[size];
		System.out.println("Enter Element of array");
		for (int i = 0; i < size; i++) {
			a[i] = sc.nextInt();
		}

		System.out.println("Enter Your Choice");
		System.out.println("1:QuickSort");
		System.out.println("2:SselectionSort");
		int c = sc.nextInt();

		iSorting d;
		long startTime, endTime, total;

		switch (c) {
		case 1:
			startTime = System.currentTimeMillis();
			d = new QSort();
			System.out.println("Using Quick sort....");
			d.sort(a);
			printSortedData(a);
			endTime = System.currentTimeMillis();

			total = (endTime - startTime); // Total execution time in nano seconds
			System.out.println("Time required " + total + " millisec");
			break;

		case 2:
			startTime = System.currentTimeMillis();
			d = new SSort();
			System.out.println("Using Selection sort....");
			d.sort(a);
			printSortedData(a);
			endTime = System.currentTimeMillis();

			total = (endTime - startTime); // Total execution time in nano seconds
			System.out.println("Time required " + total + " millisec");
			break;

		default:
			System.out.println("Enter correct value");
		}

	}

}
